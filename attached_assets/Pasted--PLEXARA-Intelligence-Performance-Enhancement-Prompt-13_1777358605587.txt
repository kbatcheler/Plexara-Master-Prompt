## ENHANCEMENT A: PERFORMANCE — Speed Up The Interpretation Pipeline

**Problem:** Blood panel interpretation takes 40-60+ seconds (extraction ~10s → 3 lenses ~25s → reconciliation ~12s). Users are waiting too long.

**The current pipeline is:**
```
Upload → Extraction (Claude, ~10s) → PII Strip → [wait]
  → Lens A + B + C in parallel (~25s) → [wait]
  → Reconciliation (Claude, ~12s) → [wait]
  → DB Transaction → Mark complete
  → setImmediate: Post-interpretation orchestrator (background)
```

### A1. Use a faster model for extraction

Extraction is a structured data extraction task — it doesn't need the full reasoning power of the reconciliation model. Switch to a faster model.

In `extraction.ts`, change every `anthropic.messages.create` call to use a dedicated extraction model:

```typescript
model: process.env.LLM_EXTRACTION_MODEL || "claude-haiku-4-5-20251001",
```

Add to `.env.example`:
```bash
# Extraction model — a faster, cheaper model is sufficient for structured
# data extraction from PDFs/images. Saves ~3-5s per upload.
LLM_EXTRACTION_MODEL=claude-haiku-4-5-20251001
```

This alone should cut extraction time from ~10s to ~3-5s.

### A2. Overlap extraction DB writes with lens dispatch

In `records-processing.ts`, the current flow is: extract → write biomarkers to DB → THEN start lenses. The biomarker DB write can overlap with lens dispatch because the lenses receive the extracted data directly (not from the DB).

Restructure the pipeline so lens dispatch starts as soon as extraction completes, while the biomarker DB insert runs concurrently:

```typescript
// Start lenses immediately with the extracted data
const lensesPromise = runLensesInParallel(anonymised, patientCtx, history);

// Write biomarkers to DB concurrently (doesn't block lenses)
const biomarkerWritePromise = writeBiomarkersToDb(patientId, recordId, structuredData);

// Wait for both
const [lensResults] = await Promise.all([lensesPromise, biomarkerWritePromise]);
```

This saves the biomarker write latency (~1-2s) from the critical path.

### A3. Add a progress polling endpoint

### A4. Cache biomarker reference data in memory

The biomarker reference table is static seed data that doesn't change at runtime. Currently it's queried from the database on every interpretation. Load it once at boot and cache it:

```typescript
// lib/biomarker-cache.ts
import { db, biomarkerReferenceTable } from "@workspace/db";

let cache: Map<string, BiomarkerRef> | null = null;

export async function getBiomarkerReference(name: string): Promise<BiomarkerRef | null> {
  if (!cache) {
    const all = await db.select().from(biomarkerReferenceTable);
    cache = new Map(all.map(r => [r.biomarkerName.toLowerCase(), r]));
  }
  return cache.get(name.toLowerCase()) ?? null;
}

export async function getAllBiomarkerReferences(): Promise<Map<string, BiomarkerRef>> {
  if (!cache) {
    const all = await db.select().from(biomarkerReferenceTable);
    cache = new Map(all.map(r => [r.biomarkerName.toLowerCase(), r]));
  }
  return cache;
}

export function invalidateBiomarkerCache() { cache = null; }
```

Replace all direct `biomarkerReferenceTable` queries in the interpretation pipeline with calls to this cache.

### B2. Build the ratio computation engine

```typescript
// ratios.ts (continued)

import { db, biomarkerResultsTable } from "@workspace/db";
import { eq, and, desc } from "drizzle-orm";

export interface ComputedRatio {
  spec: RatioSpec;
  numeratorValue: number;
  denominatorValue: number;
  ratio: number;
  status: "optimal" | "normal" | "watch" | "urgent";
  interpretation: string;
}

export async function computeRatiosForPatient(patientId: number): Promise<ComputedRatio[]> {
  // Get the latest value for each biomarker (deduplicated by name, most recent first)
  const allResults = await db.select()
    .from(biomarkerResultsTable)
    .where(eq(biomarkerResultsTable.patientId, patientId))
    .orderBy(desc(biomarkerResultsTable.createdAt));

  const latestByName = new Map<string, { value: number; unit: string | null }>();
  for (const r of allResults) {
    const key = r.biomarkerName.toLowerCase();
    if (!latestByName.has(key) && r.value) {
      const v = parseFloat(r.value);
      if (Number.isFinite(v) && v > 0) {
        latestByName.set(key, { value: v, unit: r.unit });
      }
    }
  }

  const computed: ComputedRatio[] = [];

  for (const spec of RATIO_SPECS) {
    const num = latestByName.get(spec.numerator.toLowerCase());
    const den = latestByName.get(spec.denominator.toLowerCase());
    if (!num || !den || den.value === 0) continue;

    const ratio = num.value / den.value;

    let status: ComputedRatio["status"] = "normal";
    let interpretation = spec.interpretation.optimal;

    if (spec.clinicalHigh !== null && ratio > spec.clinicalHigh) {
      status = "urgent";
      interpretation = spec.interpretation.high;
    } else if (spec.clinicalLow !== null && ratio < spec.clinicalLow) {
      status = "urgent";
      interpretation = spec.interpretation.low;
    } else if (spec.optimalHigh !== null && ratio > spec.optimalHigh) {
      status = "watch";
      interpretation = spec.interpretation.high;
    } else if (spec.optimalLow !== null && ratio < spec.optimalLow) {
      status = "watch";
      interpretation = spec.interpretation.low;
    } else {
      status = "optimal";
    }

    computed.push({ spec, numeratorValue: num.value, denominatorValue: den.value, ratio, status, interpretation });
  }

  return computed;
}
```

export function buildMedicationBlock(medications: Medication[]): string {
  if (medications.length === 0) return "";

}

  const isSummer = (latitude > 0 && testMonth >= 5 && testMonth <= 9) || (latitude < 0 && (testMonth <= 3 || testMonth >= 11));
  if (isSummer && value < 40) return "Value is below optimal even in peak sun season — suggests genuine deficiency, not seasonal.";
